function checkWorkStatus() { // Get values from the input fields const projectName = document.getElementById('project-name').value; const employeeName = document.getElementById('employee-name').value; const startDate = document.getElementById('start-date').value; const endDate = document.getElementById('end-date').value; const workStatus = document.getElementById('work-status').value; // Check if all fields have values if (projectName && employeeName && startDate && endDate && workStatus) { // Create a new row for the table const table = document.getElementById('work-status-table').getElementsByTagName('tbody')[0]; const newRow = table.insertRow(table.rows.length); // Insert cells for each data item const cell1 = newRow.insertCell(0); const cell2 = newRow.insertCell(1); const cell3 = newRow.insertCell(2); const cell4 = newRow.insertCell(3); const cell5 = newRow.insertCell(4); // Fill the cells with the input data cell1.textContent = projectName; cell2.textContent = employeeName; cell3.textContent = startDate; cell4.textContent = endDate; cell5.textContent = workStatus; // Get today's date for comparison const today = new Date().toISOString().split('T')[0]; // Get today's date in YYYY-MM-DD format // Check if the end date is overdue const isOverdue = endDate < today; // If the end date is less than today's date, it's overdue // Apply the respective color class based on the status if (workStatus === "Not Started") { cell5.classList.add("overdue"); } else if (workStatus === "In Progress") { cell5.classList.add("in-progress"); } else if (workStatus === "Completed") { cell5.classList.add("completed"); } // If the task is overdue, show a caution sign and apply a red background if (isOverdue) { cell5.innerHTML = "⚠️ Overdue"; // Add the caution sign cell5.classList.add("overdue"); } // Clear the input fields after submission document.getElementById('project-name').value = ''; // Reset to default value document.getElementById('employee-name').value = ''; // Reset to default value document.getElementById('start-date').value = ''; // Reset to default value document.getElementById('end-date').value = ''; // Reset to default value document.getElementById('work-status').value = ''; // Reset to default value } else { // Alert if any field is empty alert('Please fill in all fields!'); } }
